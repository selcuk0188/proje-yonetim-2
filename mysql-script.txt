Kullanici tableCREATE TABLE `proje_yonetim`.`kullanici` (  `id` INT NOT NULL,  `ad_soyad` VARCHAR(95) NOT NULL,  `kullanici_adi` VARCHAR(45) NOT NULL,  `sifre` VARCHAR(45) NOT NULL,  `rol_id` VARCHAR(45) NOT NULL,  `durum` VARCHAR(45) NOT NULL,  `tc_no` VARCHAR(11) NOT NULL,  PRIMARY KEY (`id`))COMMENT = 'Kullanýcý Bilgilerinin Tutulduðu Tablo';Rol tableCREATE TABLE `proje_yonetim`.`rol` (  `id` INT NOT NULL,  `rol_adi` VARCHAR(45) NULL,  `rol_yetki` VARCHAR(45) NULL COMMENT 'Rol yetkileri(sayfa ve iþlem yetkileri)\n',  `rol_tanim` VARCHAR(45) NULL,  PRIMARY KEY (`id`))COMMENT = 'Rol Bilgilerinin Tutulduðu Tablo';ProjeCREATE TABLE `proje_yonetim`.`proje` (  `id` INT NOT NULL,  `ders_id` VARCHAR(45) NOT NULL,  `proje_adi` VARCHAR(45) NOT NULL,  `durum` VARCHAR(45) NOT NULL,  `son_islem` VARCHAR(45) NOT NULL,  `son_islem_tarihi` VARCHAR(45) NOT NULL,  PRIMARY KEY (`id`))COMMENT = ' Proje Bilgilerinin Tutulduðu Tablodur';DersCREATE TABLE `proje_yonetim`.`ders` (  `id` INT NOT NULL,  `ders_adi` VARCHAR(95) NOT NULL,  PRIMARY KEY (`id`))COMMENT = 'dersler';Proje_durumCREATE TABLE `proje_yonetim`.`proje_onay` (  `id` INT NOT NULL,  `proje_id` VARCHAR(45) NOT NULL,  `durum` VARCHAR(45) NULL,  PRIMARY KEY (`id`));MesajCREATE TABLE `proje_yonetim`.`mesaj` (  `id` INT NOT NULL,  `mesaj_basligi` VARCHAR(155) NOT NULL,  `mesaj_konusu` VARCHAR(255) NOT NULL,  `mesaj_icerigi` VARCHAR(455) NOT NULL,  `yeni_mesaj_mi` VARCHAR(45) NOT NULL,  `dosya_eki_var_mi` VARCHAR(45) NULL,  `dosya_icerigi` VARCHAR(45) NOT NULL,  `gonderen_kullanici_id` VARCHAR(45) NOT NULL,  `gonderilen_kullanici_id` VARCHAR(45) NOT NULL,  `okuyan_sayisi` INT NULL,  `okumayan_sayisi` INT NULL,  `durum` VARCHAR(45) NOT NULL,  `tarih` DATE NOT NULL,  PRIMARY KEY (`id`));
